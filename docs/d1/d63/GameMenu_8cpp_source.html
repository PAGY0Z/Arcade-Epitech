54  void GameMenu::loadInterface()
55  {
56  try
57  {
58  std::string line;
59  std::ifstream file("assets/menu/map.txt");
60  if (!file.is_open())
61  throw GameMenuException("Could not open the map file for the menu", FILE_DATA);
62  while (std::getline(file, line))
63  {
64  for (std::size_t i = 0; i < line.size(); i++)
65  {
66  if (line[i] == ' ')
67  {
68  continue;
69  }
70  else if (('A' <= line[i] && line[i] <= 'Z') || ('0' <= line[i] && line[i] <= '9'))
71  {
72  _interface.push_back(AEntity(i, _screen_size_y, line[i], Orientation::UP));
73  }
74  else if (line[i] == '^' || line[i] == 'v' || line[i] == '<' || line[i] == '>')
75  {
76  if (line[i] == '^')
77  _buttons_up.push_back(AEntity(i, _screen_size_y, line[i], Orientation::UP));
78  if (line[i] == '>')
79  _buttons_right.push_back(AEntity(i, _screen_size_y, line[i], Orientation::RIGHT));
80  if (line[i] == 'v')
81  _buttons_down.push_back(AEntity(i, _screen_size_y, line[i], Orientation::DOWN));
82  if (line[i] == '<')
83  _buttons_left.push_back(AEntity(i, _screen_size_y, line[i], Orientation::LEFT));
84  }
85  }
86  if (line.size() > _screen_size_x)
87  _screen_size_x = line.size();
88  _screen_size_y++;
89  }
90  file.close();
91  }
92  catch (AException &except)
93  {
94  std::cerr << except.what() << std::endl;
95  exit(except.getReturnValue());
96  }
97  }

43  const struct DisplayData &GraphicSFML::getDisplayData() const
44  {

47  const_cast<sf::Clock *>(&_clock)->restart();
48  data->events = std::vector<AEvent>();
49  sf::Event event;
50  while (const_cast<sf::RenderWindow *>(&_window)->pollEvent(event))
51  {
52  if (event.type == sf::Event::Closed)
53  {
54  const_cast<sf::RenderWindow *>(&_window)->close();
55  }
56  else if (event.type == sf::Event::KeyPressed)
57  {
58  if (sfml_event_matcher.find(event.key.code) != sfml_event_matcher.end())
59  {
60  AEvent new_event = AEvent(sfml_event_matcher.at(event.key.code), 0, 0);
61  data->events.push_back(new_event);
62  }
63  }
64  else if (event.type == sf::Event::MouseButtonPressed)
65  {
66  if (event.mouseButton.button == sf::Mouse::Left)
67  {
68  AEvent new_event = AEvent(Type_e::MOUSE_LEFT, event.mouseButton.x / _width_factor, event.mouseButton.y / _height_factor);
69  data->events.push_back(new_event);

71  else if (event.mouseButton.button == sf::Mouse::Right)

73  AEvent new_event = AEvent(Type_e::MOUSE_RIGHT, event.mouseButton.x / _width_factor, event.mouseButton.y / _height_factor);
74  data->events.push_back(new_event);
75  }
76  }

89  update_texture_map(data.textures);

91  {
92  sf::Sprite sprite;
93  if (_textures.find(entity.getTexture()) == _textures.end())
94  throw GraphicSFMLException("Texture not found for entity '" + std::to_string(entity.getTexture()) + "'", FILE_DATA);
95  sprite.setTexture(_textures[entity.getTexture()].first);
96  sprite.setOrigin(sprite.getTexture()->getSize().x / 2, sprite.getTexture()->getSize().y / 2);
97  sprite.setRotation(static_cast<float>(entity.getOrientation()));
98  sprite.setScale((float)_width_factor / (float)sprite.getTexture()->getSize().x, (float)_height_factor / (float)sprite.getTexture()->getSize().y);
99  sprite.setPosition(entity.getPosX() * _width_factor + _width_factor / 2, entity.getPosY() * _height_factor + _height_factor / 2);
100  _window.draw(sprite);

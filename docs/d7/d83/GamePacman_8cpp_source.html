94  void GamePacman::initEntities(std::vector<std::string> map)
95  {
96  bool player_set = false;
97  int ghost_count = 0;
98  try
99  {
100  for (std::size_t y = 0; y < map.size(); y++)
101  {
102  for (std::size_t x = 0; x < map[y].size(); x++)
103  {
104  if (map[y][x] == 'p')
105  {
106  if (player_set == true)
107  throw GamePacmanException("Multiple Players in map", FILE_DATA);
108  _player = Player(x, y, map[y][x], Orientation::RIGHT);
109  player_set = true;
110  }
111  else if (map[y][x] == 'g')
112  {
113  _ghosts.push_back(Ghost(x, y, map[y][x], Orientation::UP, 10 + ghost_count * 2));
114  ghost_count++;
115  }
116  else if (map[y][x] == '.')
117  {
118  _pacgums.push_back(Pacgum(x, y, map[y][x], Orientation::UP, false));
119  }
120  else if (map[y][x] == 'o')
121  {
122  _pacgums.push_back(Pacgum(x, y, map[y][x], Orientation::UP, true));
123  }
124  else if (map[y][x] == '/' || map[y][x] == '\\' || map[y][x] == '[' || map[y][x] == ']' || map[y][x] == '|' || map[y][x] == '-')
125  {
126  _walls.push_back(Wall(x, y, map[y][x], Orientation::UP, false));
127  }
128  else if (map[y][x] == '+')
129  {
130  _walls.push_back(Wall(x, y, map[y][x], Orientation::UP, true));
131  }
132  else if (map[y][x] == '>' || map[y][x] == '<')
133  {
134  _teleporters.push_back(Teleporter(x, y, map[y][x], (map[y][x] == '>') ? Orientation::RIGHT : Orientation::LEFT));
135  }
136  else if (('A' <= map[y][x] && map[y][x] <= 'Z') || ('0' <= map[y][x] && map[y][x] <= '9'))
137  {
138  _interface.push_back(AEntity(x, y, map[y][x], Orientation::UP));
139  }
140  else
141  {
142  if (_textures.find(map[y][x]) == _textures.end())
143  throw GamePacmanException("Texture not found for entity '" + std::string(1, map[y][x]) + "'", FILE_DATA);
144  }
145  }
146  }
147  if (player_set == false)
148  throw GamePacmanException("No Player in map", FILE_DATA);
149  if (ghost_count == 0)
150  throw GamePacmanException("No Ghost in map", FILE_DATA);
151  if (_teleporters.size() != 2)
152  throw GamePacmanException("Invalid number of teleporters in map", FILE_DATA);
153  _teleporters[0].setDest(_teleporters[1].getPosX(), _teleporters[1].getPosY());
154  _teleporters[1].setDest(_teleporters[0].getPosX(), _teleporters[0].getPosY());
155  }
156  catch (const AException &except)
157  {
158  std::cerr << except.what() << std::endl;
159  exit(except.getReturnValue());
160  }
161  }

227  else if (event.getType() == Type_e::RIGHT)
228  _player.setOrientation(Orientation::RIGHT, _walls);
229  else
230  _system_events.push_back(event);
231 
232  _has_started = true;
233  if (_has_lost == true)
234  _is_running = false;

239  _has_started = false;

241  }
242  if (_has_started == false || _has_won == true || _has_lost == true)
243  {
244  delete &data;
245  return;
246  }
247  _player.move(data.delta, _walls);
248  _has_won = true;

250  if (pacgum.getIsEaten() == false)
251  {
252  _has_won = false;

254  {
255  if (pacgum.getIsSuper() == true)

261  }

264  for (auto &wall : _walls)

266  if (wall.getIsOpen() == false)
267  wall.update(data.delta);
268  for (auto &teleporter : _teleporters)
269  if (_player.collide_with(teleporter))
270  _player.teleport(teleporter);

275  if (_player.collide_with(ghost))
